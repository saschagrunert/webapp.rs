8pub async fn validate(req: Request<Body>, next: Next) -> Result<Response, StatusCode> {
9    if req.method() != Method::POST {
10        return Ok(next.run(req).await);
11    }
12
13    let host = req
14        .headers()
15        .get(header::HOST)
16        .and_then(|v| v.to_str().ok())
17        .unwrap_or_default();
18
19    let origin = req
20        .headers()
21        .get(header::ORIGIN)
22        .and_then(|v| v.to_str().ok());
23
24    let referer = req
25        .headers()
26        .get(header::REFERER)
27        .and_then(|v| v.to_str().ok());
28
29    let valid = match (origin, referer) {
30        (Some(origin), _) => origin_matches_host(origin, host),
31        (None, Some(referer)) => origin_matches_host(referer, host),
32        (None, None) => false,
33    };
34
35    if valid {
36        Ok(next.run(req).await)
37    } else {
38        Err(StatusCode::FORBIDDEN)
39    }
40}
41
42fn origin_matches_host(origin: &str, host: &str) -> bool {
43    // Extract host portion from origin URL (e.g., "http://localhost:3000" -> "localhost:3000")
44    origin
45        .split("//")
46        .nth(1)
47        .map(|h| h.split('/').next().unwrap_or(h))
48        .is_some_and(|origin_host| origin_host == host)
49}
