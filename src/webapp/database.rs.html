1use chrono::{DateTime, Utc};
2use sqlx::{PgPool, postgres::PgPoolOptions};
3use std::{env, sync::OnceLock};

5static POOL: OnceLock<PgPool> = OnceLock::new();
6
7pub async fn init() -> Result<(), sqlx::Error> {
8    let database_url =
9        env::var("DATABASE_URL").unwrap_or_else(|_| "postgres://localhost/webapp".into());
10
11    let pool = PgPoolOptions::new()
12        .max_connections(5)
13        .connect(&database_url)
14        .await?;
15
16    sqlx::migrate!().run(&pool).await?;
17
18    POOL.set(pool).expect("database pool already initialized");
19
20    Ok(())
21}
22
23pub fn pool() -> &'static PgPool {
24    POOL.get().expect("database pool not initialized")
25}

29pub async fn create_user(username: &str, password_hash: &str) -> Result<(), sqlx::Error> {
30    sqlx::query("INSERT INTO users (username, password_hash) VALUES ($1, $2)")
31        .bind(username)
32        .bind(password_hash)
33        .execute(pool())
34        .await?;
35    Ok(())
36}

38pub async fn get_password_hash(username: &str) -> Result<Option<String>, sqlx::Error> {
39    let row: Option<(String,)> =
40        sqlx::query_as("SELECT password_hash FROM users WHERE username = $1")
41            .bind(username)
42            .fetch_optional(pool())
43            .await?;
44    Ok(row.map(|r| r.0))
45}
46
47pub async fn user_exists(username: &str) -> Result<bool, sqlx::Error> {
48    let row: Option<(i32,)> = sqlx::query_as("SELECT 1 FROM users WHERE username = $1 LIMIT 1")
49        .bind(username)
50        .fetch_optional(pool())
51        .await?;
52    Ok(row.is_some())
53}

79pub async fn update_session(
80    old_token: &str,
81    new_token: &str,
82    expires_at: DateTime<Utc>,
83) -> Result<bool, sqlx::Error> {
84    let result = sqlx::query("UPDATE sessions SET token = $1, expires_at = $2 WHERE token = $3")
85        .bind(new_token)
86        .bind(expires_at)
87        .bind(old_token)
88        .execute(pool())
89        .await?;
90    Ok(result.rows_affected() > 0)
91}

93pub async fn delete_session(token: &str) -> Result<bool, sqlx::Error> {
94    let result = sqlx::query("DELETE FROM sessions WHERE token = $1")
95        .bind(token)
96        .execute(pool())
97        .await?;

101pub async fn delete_expired_sessions() -> Result<u64, sqlx::Error> {
102    let result = sqlx::query("DELETE FROM sessions WHERE expires_at < NOW()")
103        .execute(pool())

112    async fn setup() {
113        let database_url = env::var("DATABASE_URL").expect("DATABASE_URL must be set for tests");
114        let pool = PgPoolOptions::new()
115            .max_connections(5)
116            .connect(&database_url)
117            .await
118            .expect("failed to connect to test database");
119        // Drop existing tables and migration state to apply fresh schema
120        sqlx::query("DROP TABLE IF EXISTS sessions")
121            .execute(&pool)
122            .await
123            .unwrap();
124        sqlx::query("DROP TABLE IF EXISTS users")
125            .execute(&pool)
126            .await
127            .unwrap();
128        sqlx::query("DROP TABLE IF EXISTS _sqlx_migrations")
129            .execute(&pool)
130            .await
131            .unwrap();
132        sqlx::migrate!()
133            .run(&pool)
134            .await
135            .expect("failed to run migrations");
136        let _ = POOL.set(pool);
137    }
138
139    #[tokio::test]
140    async fn database_operations() {
141        setup().await;
142
143        // User creation and lookup
144        create_user("alice", "$argon2id$hash").await.unwrap();
145        assert!(user_exists("alice").await.unwrap());
146        assert!(!user_exists("bob").await.unwrap());
147
148        // Password hash retrieval
149        let hash = get_password_hash("alice").await.unwrap();
150        assert_eq!(hash.as_deref(), Some("$argon2id$hash"));
151        assert!(get_password_hash("nobody").await.unwrap().is_none());
152
153        // Session lifecycle
154        let expires = Utc::now() + chrono::Duration::hours(1);
155        create_session("tok1", "alice", expires).await.unwrap();
156        assert!(session_exists("tok1").await.unwrap());
157
158        let updated = update_session("tok1", "tok2", expires).await.unwrap();
159        assert!(updated);
160        assert!(!session_exists("tok1").await.unwrap());
161        assert!(session_exists("tok2").await.unwrap());
162
163        let deleted = delete_session("tok2").await.unwrap();
164        assert!(deleted);
165        assert!(!session_exists("tok2").await.unwrap());
166
167        // Deleting nonexistent session returns false
168        assert!(!delete_session("nonexistent").await.unwrap());
169
170        // Expired session cleanup
171        let past = Utc::now() - chrono::Duration::hours(1);
172        create_session("expired_tok", "alice", past).await.unwrap();
173        let count = delete_expired_sessions().await.unwrap();
174        assert!(count > 0);
175        assert!(!session_exists("expired_tok").await.unwrap());
176    }
