14const MAX_REQUESTS: usize = 20;
15const WINDOW: Duration = Duration::from_secs(60);
16
17static CLIENTS: Mutex<Option<HashMap<IpAddr, Vec<Instant>>>> = Mutex::new(None);
18
19fn is_allowed(ip: IpAddr) -> bool {
20    let mut guard = CLIENTS.lock().unwrap_or_else(|e| e.into_inner());
21    let map = guard.get_or_insert_with(HashMap::new);
22    let now = Instant::now();
23    // Evict IPs with no recent requests to prevent unbounded growth
24    map.retain(|_, v| {
25        v.retain(|t| now.duration_since(*t) < WINDOW);
26        !v.is_empty()
27    });
28    let timestamps = map.entry(ip).or_default();
29    if timestamps.len() >= MAX_REQUESTS {
30        false
31    } else {
32        timestamps.push(now);
33        true
34    }
35}

37fn extract_ip(req: &Request<Body>) -> IpAddr {
38    req.headers()
39        .get("x-forwarded-for")
40        .and_then(|v| v.to_str().ok())
41        .and_then(|s| s.split(',').next())
42        .and_then(|s| s.trim().parse::<IpAddr>().ok())
43        .unwrap_or(IpAddr::from([127, 0, 0, 1]))
44}
45
46pub async fn check(req: Request<Body>, next: Next) -> Result<Response, StatusCode> {
47    if req.method() != Method::POST {
48        return Ok(next.run(req).await);
49    }
50
51    if is_allowed(extract_ip(&req)) {
52        Ok(next.run(req).await)
53    } else {
54        Err(StatusCode::TOO_MANY_REQUESTS)
55    }
56}
